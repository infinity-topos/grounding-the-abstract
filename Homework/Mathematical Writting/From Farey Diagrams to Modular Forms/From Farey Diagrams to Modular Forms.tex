\section{Topology of Fractions: Farey's Diagram}
Let $\infty = 1/0$ and $\hat{\Q}=\Q \cup \{\infty\}$.

\begin{definition}[Farey Sequence]
    The Farey sequence $\{F_n\}$ is a sequence of subsets of $\mathbb{Q}$ defined by
    \begin{align*}
        F_n :=\{a/b \mid 0\le a \le b \le n \}.
    \end{align*}
\end{definition}

\begin{definition}[Negative Farey Sequence]
    The Farey sequence $\{G_n\}$ is a sequence of subsets of $\mathbb{Q}$ defined by
    \begin{align*}
        G_n :=\{-a/b \mid 0\le a < b \le n \}\cup\{1\}.
    \end{align*}
\end{definition}

\begin{definition}[Mediant Operation]
    The Mediant Operation $\oplus$ is a combinatorial rule defined by 
    \begin{align*}
        \frac{a}{c} \oplus \frac{b}{d}:= \frac{a+b}{c+d}
    \end{align*}
    for $a/c$ and $b/d$ are redueced.
\end{definition}

\begin{definition}[Negative Mediant Operation]
     The Negative Mediant Operation $\ominus$ is a combinatorial rule defined by 
    \begin{align*}
        \frac{a}{c} \ominus \frac{b}{d}:= \frac{-|a|-|b|}{c+d}
    \end{align*}
    for $a/c$ and $b/d$ are redueced.
\end{definition}

Observe that Farey sequence (resp. nagative) can be constructed by doing successively Mediant operation (resp. nagative).
For example, 
\begin{align*}
    F_2 =\{0/1 ,0/1\oplus 1/1,1/1\}. 
\end{align*}

\begin{remark}[Pythagorean Coordinate]
    The Cayley transformation on $p/q$ also can be done in the following way. 
    Assigne each $p/q$ a Pythagorean triple $(2pq,p^2-q^2,p^2+q^2)$. Consider the Projectivization of this triple. Namely, we obtain an assignment 
    \begin{align*}
        p/q \mapsto  \left(\frac{p^2-q^2}{p^2+q^2},\frac{-2pq}{p^2+q^2}\right).
    \end{align*}
    This coordinate agree on Cayley transformation.
    
    In the perspective of Pythagorean Coordinate, the reduced rational number $p/q$ also can be interpreted as a integral point of quadratic cone defined by $x^2+y^2-z^2$.
\end{remark}

\section{Modular Group}
We are interested in the automorphism group on cusp point. 
\begin{definition}[Modular Group]
    Modular Group is defined by the automorphism of projective variety
    \begin{align*}
        \text{Aut}(\mathbb{P}_{\mathbb{Z}}^1)^+ \simeq \mathrm{PSL}_2(\mathbb{Z})
    \end{align*}
\end{definition}

The modular group also gives the automorphism of ideal triangles.
From the "functor of point", we can interpreted each ideal triangles $\Delta$ as the morphisim $g:\Delta_0\to \Delta\in   \mathrm{PSL}_2(\mathbb{Z})/(\mathbb{Z}/3\mathbb{Z})$.
Hence, if we consider direction, the edge correspondence must be considerated, that is, each edge $e_i$ of an ideal triangle corresponds to the morphisim $h:e\to e_i\in \mathrm{PSL}_2(\mathbb{Z})/(\mathbb{Z}/2\mathbb{Z})$.

The modular group has two generator 
\begin{align*}
    S = \begin{pmatrix} 0 & -1 \\ 1 & 0 \end{pmatrix} \text{ and } T = \begin{pmatrix} 1 & 1 \\ 0 & 1 \end{pmatrix}
\end{align*}
Geometrically, $S$ corresponds to the "rotation of the standard triangle by an edge", namely, it represent the morphisim $\{0,1,\infty\}\to \{\infty,-1,0\}$ respectively.
$T$ represent the shift operator $\{0,1,\infty\}\to \{1,2,\infty\}$. $R=TS$ represents the cyclic permutation of cusp point $\{0,1,\infty\}\to \{1,\infty,0\}$.

\begin{proposition}[Free Product Structure of the Modular Group]
    The modular group $\Gamma = \mathrm{PSL}_2(\mathbb{Z})$ is isomorphic to the free product of two cyclic groups of orders $2$ and $3$. Explicitly,
    \begin{align*}
        \Gamma \cong \mathbb{Z}/2\mathbb{Z} * \mathbb{Z}/3\mathbb{Z}.
    \end{align*}
\end{proposition}

\begin{construction}
    Let $G = H \star K$ be a free product. We define a graph $\Gamma$ as follows.
    \begin{enumerate}
        \item The vertex set $V$ consists of two types $H$ and $K$: $V = \{gH, gK : g \in G\}$.
        \item The edge set $E$ consists of all group elements in $G$.
        \item The edge $g \in E = G$ connects $gH$ and $gK$.
    \end{enumerate}

    Then $G$ acts on $\Gamma$: each element $g \in G$ sends $xH$ to $gxH$ and $xK$ to $gxK$. The edge relation is preserved. So $G$ acts on $\Gamma$ by graph isomorphism.
\end{construction}

\begin{theorem}[Bass-Serre Trees for free products]
    The graph $\Gamma$ is a tree so that the degree of vertex of type $H$ (resp. $K$) equals $\#H$ (resp. $\#K$).

Moreover, the action of $G$ on $\Gamma$ has trivial edge stabilizers and vertices stabilizers of type $H$ and $K$ conjugated to $H$ and $K$ respectively so that the quotient is an interval.
\end{theorem}

    the Bass-Serre tree $\Gamma$ is the \textbf{incidence graph} of the dual of the Farey graph.
\end{theorem}

\section{Hyperbolic Geometry}
We aim to demonstrate that the upper half-plane and hyperbolic geometry represent the \textbf{canonical geometric realization} of this structure.
Here is our roadmap.
\begin{description}
    \item[Step 1. Algebraic Structure] 
    The modular group $G = PSL_2(\mathbb{Z})\cong C_2 \star C_3$ and its \textbf{Bass-Serre tree} $\mathcal{T}$ which is a $(2,3)$-regular bipartite graph whose vertices represent the cosets of the stabilizers $C_2$ (black nodes) and $C_3$ (white nodes).

    \item[Step 2. Gluing Riemann Surfaces] 
    Realize $\mathcal{T}$ by gluing a collection of ideal triangles. 
    At each white node (degree 3), we join the vertices of three triangles; at each black node (degree 2), we glue two triangle edges together. This assembly process yields a continuous, oriented \textbf{Riemann surface $X$}.

    \item[Step 3. Arithmetic Mapping (The Dessin d'Enfant)] 
    According to \textbf{Grothendieck's Theory of Dessins d'Enfants}, Riemann surface $X$ admits a \textbf{Belyi map} $\beta: X \to \mathbb{P}^1$. 
    This holomorphic map projects $X$ onto the Riemann sphere, ramified only over $\{0, 1, \infty\}$, such that the original tree $\mathcal{T}$ is recovered as the pre-image of the unit interval: $\mathcal{T} = \beta^{-1}([0, 1])$.

    \item[Step 4. Geometric Uniformization (The Emergence of $\mathbb{H}$)] 
    The universal cover $\mathbb{P}^1 \setminus \{0, 1, \infty\}$ is the \textbf{Hyperbolic Upper Half-Plane $\mathbb{H}$} with $\mathcal{T}$ as its symmetric geometric skeleton.
\end{description}
\begin{proposition}
    The Riemann surface $X$ deformation retracts onto Bass-Serre Tree.
\end{proposition}

We interpreted each point $\tau$ in the half-plane $\mathbb{H}$ as an elliptic curve by the construction:
\begin{align*}
    \Lambda_\tau = \{ m\tau + n \mid m,n \in \mathbb{Z} \}\quad   T_\tau = \mathbb{C} / \Lambda_\tau
\end{align*}
In this case, we call $X$ is the moduli space of elliptic curves.

\begin{definition}[Isogeny]
    Two elliptic curves $T_\tau, T_{\tau'}$ are said to be isogenous of degree $n$ if there exists $\alpha \in \mathbb{C}^\times$ such that $\alpha \Lambda_\tau \subseteq \Lambda_{\tau'}$ such that $[\Lambda_{\tau'}:\Lambda_{\tau}]=n$.
\end{definition}


Recalled that we interpreted each vertex in Bass-Serre tree as the element of $\Gamma$ with explicit form 
\begin{align*}
    \begin{pmatrix} a & b \\ c & d \end{pmatrix}
\end{align*}
such that $|ad-bc|=1$. 
Such elements represent isomorphisms of the underlying lattices.


\begin{definition}[Hecke Congruence Subgroup]
    The Hecke congruence subgroup $\Gamma_0(N)$ is the subgroup of $SL_2(\mathbb{Z})$ consisting of elements that preserve a cyclic isogeny of degree $N$. 
    Equivalently, it is defined as:
    \begin{align*}
    \Gamma_0(N) = \left\{ \begin{pmatrix} a & b \\ c & d \end{pmatrix} \in SL_2(\mathbb{Z}) \mid c \equiv 0 \pmod N \right\}
    \end{align*}
\end{definition}




\begin{definition}[Modular Curve]
    The Modular Curve is defined by $\Gamma_0(N) \backslash X$.
\end{definition}
\begin{proposition}
    The Modular Curve deformation retracts onto quotient graph $\Gamma_0(N) \backslash \mathcal{T}$.
\end{proposition}

\begin{definition}[Theta Function]
    Let $Q(x, y)$ be a positive definite integral binary quadratic form. 
    Let $R_m$ be the number of integer representations of $m$ by $Q$, that is,
    \begin{align*}
        R_m = \#\{(x,y) \in \mathbb{Z}^2 \mid Q(x,y) = m\}.
    \end{align*}
    The Theta function associated with $Q$ is defined by the generating function of the sequence 
    $\{R_m\}$:
    \begin{align*}
        \Theta_Q(\tau) = \sum_{(x,y) \in \mathbb{Z}^2} e^{2\pi i Q(x,y)\tau} = \sum_{m=0}^{\infty} R_m q^m
    \end{align*}
    where $q = e^{2\pi i \tau}$ and $\tau \in \mathbb{H}$.


\end{definition}


\usetikzlibrary{calc}


\begin{tikzpicture}[scale=1, thick]

    % --- 颜色定义 ---
    \colorlet{fareyedge}{cyan!70!blue}      % Farey 曲边三角形的测地线
    \colorlet{toponode}{red!80!black}       % Topograph 的节点
    \colorlet{topoedge}{red!70!orange}      % Topograph 的树边
    \colorlet{regiontext}{blue!70!black}    % 区域/有理数标签

    % --- 绝对边界 (Poincaré Disk Boundary) ---
    \draw[gray!40, dashed, thin] (0,0) circle (2);
    
    % --- 定义理想边界上的顶点 (Farey 尖点) ---
    % 主三角形顶点
    \coordinate (Pinf)  at (90:2);   % 1/0
    \coordinate (P0)    at (210:2);  % 0/1
    \coordinate (P1)    at (-30:2);  % 1/1
    
    % 外围邻接三角形顶点
    \coordinate (P-1)   at (150:2);  % -1/1
    \coordinate (P2)    at (30:2);   % 2/1
    \coordinate (Phalf) at (270:2);  % 1/2

    % --- 1. 绘制 Farey 图：曲边三角形 (底层图) ---
    % 填充中央主三角形作为视觉引导
    \fill[cyan!5] (Pinf) to[bend left=26] (P0) to[bend left=26] (P1) to[bend left=26] cycle;

    % 中央主三角形的三条双曲边
    \draw[fareyedge] (Pinf) to[bend left=26] (P0);
    \draw[fareyedge] (P0)   to[bend left=26] (P1);
    \draw[fareyedge] (P1)   to[bend left=26] (Pinf);

    % 外围三个邻接曲边三角形
    \draw[fareyedge] (Pinf) to[bend left=20] (P-1) to[bend left=20] (P0);
    \draw[fareyedge] (Pinf) to[bend right=20] (P2) to[bend right=20] (P1);
    \draw[fareyedge] (P0)   to[bend right=20] (Phalf) to[bend right=20] (P1);

    % --- 2. 绘制 Conway Topograph：对偶树 (上层图) ---
    % 确定各个曲边三角形的“中点” (对偶节点)
    \coordinate (C0) at (0,0);       % 主三角形中点
    \coordinate (CL) at (150:0.85);  % 左上三角形中点
    \coordinate (CR) at (30:0.85);   % 右上三角形中点
    \coordinate (CB) at (270:0.85);  % 底部三角形中点

    % 模拟更外层的 3-度正则发散树枝
    \coordinate (CL1) at (132:1.35);
    \coordinate (CL2) at (168:1.35);
    \coordinate (CR1) at (12:1.35);
    \coordinate (CR2) at (48:1.35);
    \coordinate (CB1) at (252:1.35);
    \coordinate (CB2) at (288:1.35);

    % 连接相邻三角形的中点 (正交穿过蓝色曲边)
    \draw[topoedge, very thick] (C0) -- (CL);
    \draw[topoedge, very thick] (C0) -- (CR);
    \draw[topoedge, very thick] (C0) -- (CB);

    % 绘制向外无限发散的虚线树枝
    \draw[topoedge, thick, dashed] (CL) -- (CL1);
    \draw[topoedge, thick, dashed] (CL) -- (CL2);
    \draw[topoedge, thick, dashed] (CR) -- (CR1);
    \draw[topoedge, thick, dashed] (CR) -- (CR2);
    \draw[topoedge, thick, dashed] (CB) -- (CB1);
    \draw[topoedge, thick, dashed] (CB) -- (CB2);

    % 强调显示中点 (节点)
    \fill[toponode] (C0) circle (2pt);
    \fill[toponode] (CL) circle (2pt);
    \fill[toponode] (CR) circle (2pt);
    \fill[toponode] (CB) circle (2pt);

    % --- 3. 添加数学符号与对偶字典映射标签 ---
    % 顶点对应的有理数
    \node[above, font=\small, regiontext]       at (Pinf)  {$1/0$};
    \node[below left, font=\small, regiontext]  at (P0)    {$0/1$};
    \node[below right, font=\small, regiontext] at (P1)    {$1/1$};
    \node[above left, font=\footnotesize, regiontext]  at (P-1)   {$-1/1$};
    \node[above right, font=\footnotesize, regiontext] at (P2)    {$2/1$};
    \node[below, font=\footnotesize, regiontext]       at (Phalf) {$1/2$};

    % 在 Topograph 中，原图顶点对应的是被树枝环绕的“区域 (Region)”
    \node[regiontext, font=\footnotesize] at (90:1.3)  {};
    \node[regiontext, font=\footnotesize] at (210:1.3) {};
    \node[regiontext, font=\footnotesize] at (-30:1.3) {};

\end{tikzpicture}

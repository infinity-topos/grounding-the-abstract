\begin{tikzpicture}[
    scale=1.7,
    line cap=round,
    line join=round,
    every node/.style={font=\scriptsize, inner sep=1pt}
]
\def\R{1.75}

%--------------------------------
% boundary points on the circle
% real number t is sent to angle 2 arctan(t)-90
%--------------------------------
\newcommand{\silentcusp}[2]{%
  \pgfmathsetmacro{\angtmp}{2*atan(#2)-90}%
  \expandafter\xdef\csname ang@#1\endcsname{\angtmp}%
  \coordinate (#1) at ({\angtmp}:\R);
}

\newcommand{\labeledcusp}[4]{%
  \pgfmathsetmacro{\angtmp}{2*atan(#2)-90}%
  \expandafter\xdef\csname ang@#1\endcsname{\angtmp}%
  \coordinate (#1) at ({\angtmp}:\R);
  \node[#4] at ({\angtmp}:1.08*\R) {$#3$};
}

% geodesic in the disk model:
% circle orthogonal to the boundary circle
\newcommand{\geodesic}[2]{%
  \pgfmathsetmacro{\A}{\csname ang@#1\endcsname}%
  \pgfmathsetmacro{\B}{\csname ang@#2\endcsname}%
  \pgfmathsetmacro{\m}{(\A+\B)/2}%
  \pgfmathsetmacro{\d}{abs(\B-\A)/2}%
  \pgfmathsetmacro{\c}{\R/cos(\d)}%
  \pgfmathsetmacro{\rho}{\R*tan(\d)}%
  \begin{scope}
    \clip (0,0) circle (\R);
    \draw[line width=0.35pt] ($(0,0)+(\m:\c)$) circle (\rho);
  \end{scope}
}

%--------------------------------
% outer boundary
%--------------------------------
\draw[line width=0.5pt] (0,0) circle (\R);

%--------------------------------
% cusps
%--------------------------------
% infinity
\coordinate (inf) at (90:\R);
\expandafter\xdef\csname ang@inf\endcsname{90}
\node[above] at (90:1.08*\R) {$\infty$};

% labeled cusps
\labeledcusp{mthree}{-3}{-3/1}{left}
\labeledcusp{mtwo}{-2}{-2/1}{left}
\labeledcusp{mone}{-1}{-1/1}{left}
\labeledcusp{mhalf}{-0.5}{-1/2}{below left}
\labeledcusp{zero}{0}{0/1}{below}
\labeledcusp{half}{0.5}{1/2}{below right}
\labeledcusp{one}{1}{1/1}{right}
\labeledcusp{two}{2}{2/1}{right}
\labeledcusp{three}{3}{3/1}{right}

% extra unlabeled cusps to make the figure denser
\silentcusp{mthreetwo}{-1.5}
\silentcusp{mtthird}{-0.6666667}
\silentcusp{mthird}{-0.3333333}
\silentcusp{third}{0.3333333}
\silentcusp{tthird}{0.6666667}
\silentcusp{threetwo}{1.5}

% draw cusp points
\foreach \P in {inf,mthree,mtwo,mone,mthreetwo,mtthird,mhalf,mthird,zero,third,half,tthird,one,threetwo,two,three}
  \fill (\P) circle (0.55pt);

%--------------------------------
% main Farey edges
%--------------------------------

% edges from infinity to integers
\draw[line width=0.35pt] (inf) -- (zero); % diameter
\geodesic{inf}{mthree}
\geodesic{inf}{mtwo}
\geodesic{inf}{mone}
\geodesic{inf}{one}
\geodesic{inf}{two}
\geodesic{inf}{three}

% edges from 0 to reciprocals
\geodesic{zero}{mone}
\geodesic{zero}{mhalf}
\geodesic{zero}{mthird}
\geodesic{zero}{third}
\geodesic{zero}{half}
\geodesic{zero}{one}

% neighboring finite cusps (gives the small boundary arcs)
\geodesic{mthree}{mtwo}
\geodesic{mtwo}{mone}
\geodesic{mone}{mhalf}
\geodesic{mhalf}{zero}
\geodesic{zero}{half}
\geodesic{half}{one}
\geodesic{one}{two}
\geodesic{two}{three}

% extra edges for a denser tessellation
\geodesic{mthreetwo}{mtwo}
\geodesic{mthreetwo}{mone}
\geodesic{mtthird}{mone}
\geodesic{mtthird}{mhalf}
\geodesic{mthird}{mhalf}
\geodesic{mthird}{zero}

\geodesic{third}{zero}
\geodesic{third}{half}
\geodesic{tthird}{half}
\geodesic{tthird}{one}
\geodesic{threetwo}{one}
\geodesic{threetwo}{two}

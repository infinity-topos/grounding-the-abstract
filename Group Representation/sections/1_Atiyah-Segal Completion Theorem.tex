
\section{Atiyah-Segal Completion Theorem}


\begin{theorem}[Atiyah-Segal Completion Theorem]
Let $G$ be a compact Lie group and $X$ be a compact $G$-space. Consider the projection map $EG \times X \to X$, which induces a natural homomorphism $\alpha: K_G(X) \to K(EG \times_G X)$ by regarding a $G$-bundle over $X$ as a bundle over the homotopy quotient $X_G$.

The map $\alpha$ induces an isomorphism between the completion of the equivariant K-theory with respect to the augmentation ideal $I(G)$ and the ordinary K-theory of the homotopy quotient:
\begin{equation}
    \alpha: \widehat{K_G(X)}_{I(G)} \xrightarrow{\cong} K(EG \times_G X).
\end{equation}
In particular, for $X = \text{pt}$, we have an isomorphism $\widehat{R(G)}_{I(G)} \cong K(BG)$.
\end{theorem}
